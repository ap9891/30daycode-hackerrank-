#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    double mc;
    int tip,tax;
    cin>>mc>>tip>>tax;
    cout<<(double)mc<<endl;
    float t=0;
    t = (double)mc*(tip/100.0);
    cout<<t<<endl;
    float x=0;
    x = mc*(tax/100.0);
    cout<<x<<endl;
    double total=0;
    total = mc+t+x;
    cout<<total<<endl;
    cout<<round(total);
    return 0;
}
